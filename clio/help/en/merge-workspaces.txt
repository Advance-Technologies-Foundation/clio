NAME
    merge-workspaces - Merge packages from multiple workspaces and install them to the environment

USAGE
    clio merge-workspaces [options]

DESCRIPTION
    Merge packages from multiple workspaces and install them to the environment.

ALIASES
    mergew

EXAMPLES
    clio merge-workspaces -e dev

OPTIONS
    --workspaces <VALUE>
        Comma-separated list of workspace paths to merge. Required.
    --output <VALUE>
        Path where to save the merged ZIP file. If not specified, ZIP will not be
        saved
    --name <VALUE>
        Name for the resulting ZIP file (without .zip extension). Default:
        MergedCreatioPackages.
    --install
        Whether to install the merged packages into Creatio. Default: True.

ENVIRONMENT OPTIONS
    -u, --uri <VALUE>
        Application uri
    -p, --Password <VALUE>
        User password
    -l, --Login <VALUE>
        User login (administrator permission required)
    -i, --IsNetCore
        Use NetCore application
    -e, --Environment <VALUE>
        Environment name
    -m, --Maintainer <VALUE>
        Maintainer name
    -c, --dev <VALUE>
        Developer mode state for environment
    --WorkspacePathes <VALUE>
        Workspace path
    -s, --Safe <VALUE>
        Safe action in this environment
    --clientId <VALUE>
        OAuth client id
    --clientSecret <VALUE>
        OAuth client secret
    --authAppUri <VALUE>
        OAuth app URI
    --silent
        Use default behavior without user interaction
    --restartEnvironment
        Restart environment after execute command
    --db-server-uri <VALUE>
        Db server uri
    --db-user <VALUE>
        Database user
    --db-password <VALUE>
        Database password
    --backup-file <VALUE>
        Full path to backup file
    --db-working-folder <VALUE>
        Folder visible to db server
    --db-name <VALUE>
        Desired database name
    --force
        Force restore
    --callback-process <VALUE>
        Callback process name
    --ep <VALUE>
        Path to the application root folder
